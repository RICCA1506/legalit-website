  <script>
    document.addEventListener("DOMContentLoaded", () => {
      const logoContainer = document.getElementById("logo-container");
      const paths = document.querySelectorAll(".draw-me");

      // Init lunghezze path (necessario per lo stroke-dasharray)
      paths.forEach(p => p.style.setProperty("--path-length", p.getTotalLength()));

      let started = false;
      function startOnce() {
        if (started) return;
        started = true;

        const reduceMotion = window.matchMedia &&
          window.matchMedia("(prefers-reduced-motion: reduce)").matches;

        logoContainer.style.visibility = "visible";

        if (reduceMotion) {
          // Stato finale immediato (senza animazione)
          paths.forEach(p => {
            p.style.strokeDashoffset = 0;
            p.style.fillOpacity = 1;
          });
          return;
        }

        paths.forEach(p => p.classList.add("animate-svg"));
      }

      // Trigger quando entra in viewport
      if ("IntersectionObserver" in window) {
        const io = new IntersectionObserver((entries) => {
          for (const entry of entries) {
            if (entry.isIntersecting) {
              startOnce();
              io.unobserve(entry.target);
            }
          }
        }, { threshold: 0.35 });

        io.observe(logoContainer);
      } else {
        startOnce();
      }
    });
  </script>
