'''jsx
import React, { useRef } from "react";
import gsap from "gsap";
import { useGSAP } from "@gsap/react";

const Preloader = ({ onComplete }) => {
  const containerRef = useRef(null);
  const logoRef = useRef(null);
  const wrapperRef = useRef(null);

  useGSAP(() => {
    const tl = gsap.timeline({
      onComplete: () => {
        if (onComplete) onComplete();
      }
    });

    // Initial state
    gsap.set(containerRef.current, { zIndex: 9999 });

    tl.to(logoRef.current, {
      scale: 100, // MASSIVE ZOOM
      duration: 2,
      ease: "expo.inOut",
      transformOrigin: "center center", // Zoom into the center
    })
    .to(wrapperRef.current, {
      opacity: 0,
      duration: 0.5,
      ease: "power2.inOut"
    }, "-=0.8"); // Start fading out the black wrapper while zooming finishes

  }, { scope: containerRef });

  return (
    <div ref={containerRef} className="fixed inset-0 w-full h-full pointer-events-none">
      {/* Wrapper Nero che copre tutto */}
      <div 
        ref={wrapperRef}
        className="absolute inset-0 bg-[#050505] flex items-center justify-center overflow-hidden"
      >
        {/* SVG Logo - Must be SVG for infinite scaling sharpness */}
        <svg 
          ref={logoRef}
          viewBox="0 0 200 100" 
          className="w-[200px] md:w-[400px] fill-[#C4F934]" // Colore Giallo Neon

};

export default Preloader;
