Crea un progetto web (HTML, CSS, JS) per un componente "Infinite Double Bar Reveal".

Requisiti visivi:
1. Deve sembrare una "doppia linea" gialla neon (stile F1/Lando Norris).
2. Deve scorrere da Sinistra a Destra all'infinito senza interruzioni.
3. Deve attivarsi solo quando l'utente fa scroll fino all'elemento.

Usa esattamente questa struttura di file:

<!DOCTYPE html>
<html lang="it">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Lando Style Animation</title>
    <link rel="stylesheet" href="style.css">
</head>
<body>

    <div style="height: 100vh; display: flex; align-items: center; justify-content: center; background: #111; color: #555;">
        Scrolla in basso ↓
    </div>

    <section class="content-area">
        <h2 class="title">ALWAYS BRINGING THE FIGHT.</h2>
        
        <div class="animated-underline" id="trigger-target">
            <div class="bar bar-1"></div>
            <div class="bar bar-2"></div>
        </div>
        
        <p class="subtitle">Official Merchandise Collection</p>
    </section>

    <div style="height: 50vh; background: #111;"></div>
    
    <script src="script.js"></script>
</body>
</html>

/* FILE: style.css */
body {
    margin: 0;
    padding: 0;
    background-color: #000;
    color: white;

    overflow-x: hidden;

    background: #111;

.title {
    font-size: 3rem;
    text-transform: uppercase;
    margin-bottom: 0.5rem;
    letter-spacing: -1px;
}

.subtitle {
    color: #888;
    margin-top: 1rem;
    font-family: sans-serif;
}

/* --- LOGICA ANIMAZIONE --- */

.animated-underline {
    position: relative;
    width: 100%;
    height: 12px; /* Altezza totale della doppia barra */
    overflow: hidden; /* Nasconde ciò che esce dai bordi */
    background: transparent;
}

.bar {
    position: absolute;
    top: 0;
    width: 100%;
    height: 100%;
    
    /* Disegna la DOPPIA BARRA usando un gradiente lineare */
    /* Giallo Neon: #D4F30E */
    background: linear-gradient(
        to bottom,
        #D4F30E 0%, 
        #D4F30E 30%, 
        transparent 30%, 
        transparent 70%, 
        #D4F30E 70%, 
        #D4F30E 100%
    );
    
    /* Stato iniziale: fermo */
    will-change: transform;
}

/* Posizionamento per il loop infinito (Seamless Loop) */
.bar-1 {
    left: 0;
}

.bar-2 {
    left: -100%; /* Parte da sinistra fuori schermo */
}

/* Classe che attiva l'animazione */
.animated-underline.active .bar {
    animation: slideRight 2s linear infinite;
}

@keyframes slideRight {
    0% {
        transform: translateX(0%);
    }
    100% {
        transform: translateX(100%);
    }
}

/* Responsive */
@media (max-width: 768px) {
    .title { font-size: 2rem; }
}

/* FILE: script.js */
document.addEventListener("DOMContentLoaded", () => {
    const target = document.getElementById('trigger-target');

    const observer = new IntersectionObserver((entries) => {
        entries.forEach(entry => {
            if (entry.isIntersecting) {
                // Aggiungi classe per far partire l'animazione
                entry.target.classList.add('active');
            } else {
                // Opzionale: Rimuovi se vuoi che si fermi quando esce dallo schermo (risparmia CPU)
                entry.target.classList.remove('active');
            }
        });
    }, {
        threshold: 0.2 // Attiva quando il 20% dell'elemento è visibile
    });

    if(target) observer.observe(target);
});
