  <script>
    document.addEventListener("DOMContentLoaded", () => {
      const logoWrapper = document.getElementById("logoComponent");
      const paths = document.querySelectorAll(".draw-me");

      // 1) Misura linee (necessario per stroke-dasharray/dashoffset)
      paths.forEach(path => {
        const length = path.getTotalLength();
        path.style.setProperty("--path-length", length);
      });

      // 2) Animazione autonoma: accelera a lungo, poi frena forte (morbida)
      function easeAccelThenHardBrake(t) {
        // t in [0,1]
        const split = 0.88; // più alto = accelera più a lungo
        if (t <= 0) return 0;
        if (t >= 1) return 1;

        if (t < split) {
          const u = t / split;
          // ease-in cubica (accelerazione morbida)
          return split * Math.pow(u, 3);
        } else {
          const u = (t - split) / (1 - split);
          // ease-out quintica (decelerazione più “di botto” ma fluida)
          return split + (1 - split) * (1 - Math.pow(1 - u, 5));
        }
      }

      let started = false;

      function startAutoOnce() {
        if (started) return;
        started = true;

        const duration = 2200; // ms
        const t0 = performance.now();

        function frame(now) {
          const t = Math.min(1, (now - t0) / duration);
          const p = easeAccelThenHardBrake(t);

          logoWrapper.style.setProperty("--progress", p.toFixed(4));

          if (p > 0.97) logoWrapper.classList.add("filled");

          if (t < 1) {
            requestAnimationFrame(frame);
          } else {
            logoWrapper.style.setProperty("--progress", "1");
            logoWrapper.classList.add("filled");
          }
        }

        requestAnimationFrame(frame);
      }

      // Parte al primo scroll (una sola volta)
      window.addEventListener("scroll", startAutoOnce, { once: true, passive: true });
      window.addEventListener("wheel", startAutoOnce, { once: true, passive: true });
      window.addEventListener("touchmove", startAutoOnce, { once: true, passive: true });
    });
  </script>
