/* FILE: script.js */
document.addEventListener("DOMContentLoaded", () => {
    const textElement = document.getElementById("animated-text");
    const replayBtn = document.getElementById("replay-btn");
    
    // 1. Funzione Custom SplitText (Sostituisce il plugin a pagamento)
    function splitTextIntoSpans(element) {
        const text = element.innerText;
        element.innerHTML = ""; // Pulisce il contenuto attuale
        
        const chars = text.split("");
        chars.forEach(char => {
            const span = document.createElement("span");
            span.innerText = char;
            span.className = "char";
            element.appendChild(span);
        });
        
        return element.querySelectorAll(".char");
    }

    // Preparazione iniziale
    let chars = splitTextIntoSpans(textElement);

    // 2. Funzione di animazione GSAP
    function playAnimation() {
        // Reset stato iniziale
        gsap.set(chars, { 
            y: 20, 
            opacity: 0 
        });

        // Esegui animazione
        gsap.to(chars, {
            duration: 0.8,
            y: 0,
            opacity: 1,
            stagger: 0.05,
            ease: "power2.inOut"
        });
    }

    // 3. Avvio automatico
    playAnimation();

    // 4. Gestione Replay
    replayBtn.addEventListener("click", () => {
        // Uccidi animazioni attive per evitare conflitti
        gsap.killTweensOf(chars);
        playAnimation();
    });
});
