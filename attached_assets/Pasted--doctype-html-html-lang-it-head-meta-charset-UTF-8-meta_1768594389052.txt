  <script>
    document.addEventListener("DOMContentLoaded", () => {
      const logoContainer = document.getElementById("logo-container");
      const paths = document.querySelectorAll(".draw-me");

      paths.forEach(p => p.style.setProperty("--path-length", p.getTotalLength()));

      let started = false;
      function startOnce() {
        if (started) return;
        started = true;

        logoContainer.style.visibility = "visible";
        paths.forEach(p => p.classList.add("animate-svg"));
      }

      window.addEventListener("scroll", startOnce, { once: true, passive: true });
      window.addEventListener("wheel", startOnce, { once: true, passive: true });
      window.addEventListener("touchmove", startOnce, { once: true, passive: true });
    });
  </script>
